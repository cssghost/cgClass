  • cgClass.ModulePopupLogin (options)
    登录弹出框 demo
    参数：
    •   options {Object} 参数对象数据
    •   options.addClass {css class} 登陆框的样式名称
    •   options.url {String} 登录接口
    •   options.dataType {String} 回调数据类型
    •   options.autoShow {Boolean} 是否默认弹出
    •   options.callback {Function} 登录接口的回调函数
    Since
    version 0.1
    示例：
    cgClass.Create(
	"ModulePopupLogin",
	{
  [可选]addClass : "module-popup-login",
		url : "url",
		dataType : "json",
  [可选]autoShow : true,
		callback : function(opt){
			// opt为cgClass.Popop的外部提供参数
			// 关闭方式为opt.close()
		}
	}
);
    var oLogin = cgClass.Create(
	"ModulePopupLogin",
	{
  [可选]addClass : "module-popup-login",
		url : "url",
		dataType : "json",
		autoShow : false,
		callback : function(opt){
			// opt为cgClass.Popop的外部提供参数
			// 关闭方式为opt.hide()
		}
	}
	oLogin.show();
);
