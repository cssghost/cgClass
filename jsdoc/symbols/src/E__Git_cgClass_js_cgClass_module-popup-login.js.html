1 /**
  2  * @author 徐晨 
  3  * @name cgClass.ModulePopupLogin
  4  * @class 登录弹出框 <a href="../demo/popup-login.html" target="_blank">demo</a>
  5  * @constructor
  6  * @extends cgClass.Popup
  7  * @extends jQuery
  8  * @since version 0.1 
  9  * @param {Object} options 参数对象数据
 10  * @param {css class} options.addClass 登陆框的样式名称
 11  * @param {String} options.url 登录接口
 12  * @param {String} options.dataType 回调数据类型
 13  * @param {Boolean} options.autoShow 是否默认弹出
 14  * @param {Function} options.callback 登录接口的回调函数
 15  * @example cgClass.Create(
 16 	"ModulePopupLogin",
 17 	{
 18   [可选]addClass : "module-popup-login",
 19 		url : "url",
 20 		dataType : "json",
 21   [可选]autoShow : true,
 22 		callback : function(opt){
 23 			// opt为cgClass.Popop的外部提供参数
 24 			// 关闭方式为opt.close()
 25 		}
 26 	}
 27 ); 
 28  * @example var oLogin = cgClass.Create(
 29 	"ModulePopupLogin",
 30 	{
 31   [可选]addClass : "module-popup-login",
 32 		url : "url",
 33 		dataType : "json",
 34 		autoShow : false,
 35 		callback : function(opt){
 36 			// opt为cgClass.Popop的外部提供参数
 37 			// 关闭方式为opt.hide()
 38 		}
 39 	}
 40 	oLogin.show();
 41 ); 
 42  */
 43 cgClass.AddClass(
 44 	"ModulePopupLogin",
 45 	{
 46 		init : function(options){
 47 			var self = this,
 48 				option = $.extend(/** @lends cgClass.ModulePopupLogin.prototype*/{
 49 					/**
 50 			         * 登陆框的样式名称
 51 			         * @type css class
 52 			         * @default {}
 53 			         */
 54 					addClass : "module-popup-login",
 55 					/**
 56 			         * 登录接口
 57 			         * @type String
 58 			         * @default {}
 59 			         */
 60 					url : "url",
 61 					/**
 62 			         * 回调数据类型
 63 			         * @type String
 64 			         * @default {}
 65 			         */
 66 					dataType : "json",
 67 					/**
 68 			         * 是否默认弹出
 69 			         * @type Boolean
 70 			         * @default {}
 71 			         */
 72 					autoShow : true,
 73 					/**
 74 			         * 登录接口的回调函数
 75 			         * @type Function
 76 			         * @default null
 77 			         */
 78 					callback : null
 79 				}, options),
 80 				loginPopupCon = '<div class="text Js-name"><input type="text" placeholder="请输入用户名" /><p class="error">请输入用户名</p></div>'+
 81 						'<div class="text Js-password"><input type="password" placeholder="请输入密码" /><p class="error">请输入密码</p></div>'+
 82 						'<div class="remember"><label><input type="checkbox" /> 记住密码</label></div>',
 83 				targetContent = option.content,
 84 				popupOption = $.extend(option, {
 85 					type: "popup",
 86 					title: "用户登录",
 87 					template : loginPopupCon,
 88 					content : function(opt){
 89 						if ( typeof targetContent == "function" ) {
 90 							targetContent(opt);
 91 						}
 92 						opt.oCon.on("blur", ".text input", function(){
 93 							self.testLoginNull($(this));
 94 						});
 95 					},
 96 					done : function(opt){
 97 						var $con = opt.oCon,
 98 							$text = $con.find(".text"),
 99 							$name = $con.find(".Js-name input"),
100 							$psw = $con.find(".Js-password input");
101 						opt.removeTip();
102 						if ( $.trim( $name.val() ) == "" || $.trim( $psw.val() ) == "" ) {
103 							self.testLoginNull($name);
104 							self.testLoginNull($psw);
105 							opt.disableBtn(true);
106 						}
107 						else{
108 							$con.find(".error").slideUp();
109 							$.ajax({
110 								url : option.url,
111 								type : "post",
112 								dataType : option.dataType,
113 								data : { username : $name.val(), password : $psw.val() },
114 								success : function(result){
115 									if ( typeof option.callback == "function" ) {
116 										option.callback(result, opt);
117 									}
118 									// if ( result.success ) {
119 									// 	opt.close();
120 									// } else{
121 									// 	opt.disableBtn(true);
122 									// 	opt.showTip("用户名或者密码错误");
123 									// }
124 								},
125 								error : function(){
126 									opt.showTip("登录失败，请重新尝试");
127 								}
128 							});
129 						}
130 					},
131 					cancel : function(opt){
132 						opt.hide();
133 					}
134 				});
135 			self.popup = cgClass.Create("Popup", popupOption);
136 		},
137 		/**
138 		 * @name cgClass.ModulePopupLogin#testLoginNull
139 		 * @desc  验证输入框是否为空
140 		 * @event
141 		 */
142 		testLoginNull : function($text){
143 			if ( $.trim( $text.val() ) == "" ) {
144 				$text.siblings(".error").slideDown("fast").closest(".text").addClass("text-error");
145 			}else{
146 				$text.siblings(".error").slideUp("fast").closest(".text").removeClass("text-error");
147 			}
148 		},
149 		/**
150 		 * @name cgClass.ModulePopupLogin#show
151 		 * @desc  显示弹出框
152 		 * @event
153 		 */
154 		show : function(callback){
155 			this.popup.reset();
156 			this.popup.show(callback);
157 		}
158 	}
159 );
