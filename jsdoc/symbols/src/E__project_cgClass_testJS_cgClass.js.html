 11 /** 
 12 * @author xu chen 
 13 * @constructor BaseClass 
 14 * @return {Object} 基础类
 15 * @description 生成基础js类 
 16 * @example new cgClass.BaseClass(parentClass); 
 17 * @since version 0.1 
 18 * @param {Object} parent parentClass 
 19 */ 
 20 cgClass.BaseClass = function(parent){
 21     var _class = function(){
 22         this.init.apply(this, arguments);
 23     };
 24     // Inherit
 25     if ( parent ) {
 26     	var subClass = function(){};
 27     	subClass.prototype = parent.prototype;
 28     	_class.prototype = new subClass;
 29     }
 30     _class.prototype.init = function(){};
 31     _class.fn = _class.prototype;
 32     _class.fn.parent = _class;
 33 	_class._super = _class.__proto__;
 34 
 35 	_class.attrs = function(obj){
 36 		var included = obj.included;
 37 		for( var i in obj ){
 38 			_class[i] = obj[i];
 39 		}
 40 		if ( included ) {
 41 			included( _class );
 42 		}
 43 	}
 44 
 45 	_class.extend = function(obj){
 46 		var extended = obj.extended;
 47 		for( var i in obj ){
 48 			_class.fn[i] = obj[i];
 49 		}
 50 		if ( extended ) {
 51 			extended( _class );
 52 		}
 53 	}
 54 
 55     return _class;
 56 };
