257 		parseVer : function(data, val, dom){
258 			var self = this,
259 				verData = data && data.match(/(^.*)\:(\S+)/),
260 				name, vers, result, flag = false;
261 			if ( !!verData ) {
262 				name = verData[1];
263 				vers = verData[2].split("/");
264 				if ( $.trim(val) == "" ) {
265 					if( $.inArray("notNull", vers) != -1 ){
266 						self.thrown(dom, name + "不能为空");
267 					}else{
268 						self.verified(dom);
269 					}
270 				}else{
271 					for(var i = 0; i < vers.length; i++){
272 						result = self.match(val, name, vers[i], dom);
273 						if ( result != "ajax" && !result ){
274 							break;
275 						}
276 					}
277 				}
278 			}
279 			// else{
280 			// 	alert("验证信息配置有误，请检查代码");
281 			// }

283 		match : function(val, name, ver, dom){
284 			var self = this,
285 				_ver = ver.match(/^([^\,]+)\,\{?([^\}]+)\}?$/), // 匹配"ver,({)xxx(})" ==> [1]:ver, [2]:xxx
286 				term = !!_ver ? _ver[1] : ver,
287 				mapTerm = self.map[term],
288 				msg = "", doTest;
289 			// 验证条件存在时
290 			if ( !!mapTerm) {
291 				msg = mapTerm.msg && mapTerm.msg.replace("{name}", name);
292 				// 正则验证时
293 				if ( !!mapTerm.reg ) {
294 					if ( mapTerm.reg.test(val) ) {
295 						self.verified(dom);
296 						return true;
297 					}else{
298 						self.thrown(dom, msg);
299 						return false;
300 					}
301 				}
302 				// 函数验证时
303 				else if ( !!self[term] ) {
304 					if ( !!_ver ) {
305 						doTest = self[term](val, msg, dom, _ver[2]);
306 					}else{
307 						doTest = self[term](val, msg, dom, term);
308 					}
309 					if ( !!doTest.result ) {
310 						doTest.result != "ajax" && self.verified(dom);
311 					}else{
312 						self.thrown(dom, doTest.msg);
313 					}
314 					return doTest.result;
315 				}else{
316 					// 附加函数验证
317 					if ( !!term.fun ) {
318 						doTest = term.fun(val, msg, dom, term);
319 						if ( !!doTest.result ) {
320 							doTest.result != "ajax" && self.verified(dom);
321 						}else{
322 							self.thrown(dom, doTest.msg);
323 						}
324 					}
325 				}
326 			}
327 			else{
328 				alert("验证信息配置有误，请检查代码");
329 			}
