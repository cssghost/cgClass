 11 /** 
 12 * @author xu chen 
 13 * @return {Bool} 是否在目标数组中存在
 14 * @description 判断是否在目标数组中存在，如果存在，触发回调函数
 15 * @example cgTools.isInArray(array, value, callback); 
 16 * @since version 0.1 
 17 * @param {Array} arr 目标数组
 18 * @param {param} value 查询值
 19 * @param {Function} callback 回调函数 参数为查询值的下标
 20 */ 
 21 cgTools.isInArray = function(arr, value, callback){
 22 	var result = false;
 23 	if ( arr && arr.constructor && arr.length ) {
 24 		for( var i = 0; i < arr.length; i++ ){
 25 			if ( arr[i] == value ) {
 26 				result = true;
 27 				if ( typeof callback == "function" ) {
 28 					callback(i);
 29 				}
 30 				break;
 31 			}
 32 		}
 33 	}
 34 	return result;
 35 }

 37 /** 
 38 * @author xu chen 
 39 * @return {Bool} 是否在目标数据对象中存在
 40 * @description 判断是否在目标数据对象中存在，如果存在，触发回调函数
 41 * @example cgTools.isInObject(obj, key, value, callback); 
 42 * @since version 0.1 
 43 * @param {Object} obj 目标数据对象
 44 * @param {String} key 键名
 45 * @param {param} value 键值
 46 * @param {Function} callback 回调函数 参数为key, value
 47 */ 
 48 cgTools.isInObject = function(obj, key, value, callback){
 49 	var result = false;
 50 	if ( obj[key] == value ) {
 51 		result = true;
 52 		if ( typeof callback == "function" ) {
 53 			callback(key, value);
 54 		}
 55 	}
 56 	return result;
 57 }
