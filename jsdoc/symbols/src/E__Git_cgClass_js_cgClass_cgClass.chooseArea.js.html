108 								}
109 							break;
110 							case keys.enter:
111 								$list.hide();
112 								if ( typeof( self.option.action ) == "function" ) {
113 									if ( $cur.length ) {
114 										$cur.click();
115 									} else{
116 										self.option.action(self.option, $.trim( $text.val() ), $cur );
117 									}
118 								}
119 							break;
120 							default :
121 							break;
122 						}
123 					}
124 				}else{
