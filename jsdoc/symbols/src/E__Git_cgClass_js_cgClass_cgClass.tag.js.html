  9 cgClass.AddClass(
 10 	"Tag",
 11 	{
 12 		init : function(options){
 13 			var self = this,
 14 				option = $.extend({
 15 					wrap : $(".wrap"),
 16 					tagList : ".tagList",
 17 					tagItem : ".item",
 18 					conList : ".conList",
 19 					conItem : ".conItem",
 20 					selectClass : "selected",
 21 				}, options);
 22 			option.wrap.find(option.tagList).on("click", option.tagItem, function(){
 23 				var $tag = $(this),
 24 					_index = $(this).index();
 25 				if ( !$tag.hasClass(option.selectClass) ) {
 26 					$tag.addClass(option.selectClass).siblings().removeClass(option.selectClass);
 27 					option.wrap.find(option.conList).find(option.conItem).eq(_index).show().siblings().hide();
 28 				}
 29 			});
 30 		}
 31 	}
 32 );
