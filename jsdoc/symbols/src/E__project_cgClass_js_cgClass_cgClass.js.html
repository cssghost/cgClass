 20 /**
 21  * @author 徐晨 
 22  * @name BaseClass
 23  * @class 创建基础js类
 24  * @constructor
 25  * @extends cgClass
 26  * @since version 0.1 
 27  * @param {Object} parent 可选参数，被继承类
 28  * @example var newClass = new cgClass.BaseClass(parentClass); 
 29  */
 30 cgClass.BaseClass = function(parent){
 31     var _class = function(){
 32         this.init.apply(this, arguments);
 33     };
 34     // Inherit
 35     if ( parent ) {
 36     	var subClass = function(){};
 37     	subClass.prototype = parent.prototype;
 38     	_class.prototype = new subClass;
 39     }
 40     _class.prototype.init = function(){};
 41     _class.fn = _class.prototype;
 42     _class.fn.parent = _class;
 43 	_class._super = _class.__proto__;
 44 
 45 
 46 	/**
 47 	 * @name BaseClass#attrs
 48 	 * @desc  为实体类添加属性
 49 	 * @event
 50 	 * @param {Object} obj 需要添加的属性
 51 	 * @example 
 52 	 * newClass.attrs({
 53 	 *     key : "value"
 54 	 * }); 
 55 	 */
 56 	_class.attrs = function(obj){
 57 		var included = obj.included;
 58 		for( var i in obj ){
 59 			_class[i] = obj[i];
 60 		}
 61 		if ( included ) {
 62 			included( _class );
 63 		}
 64 	}
 65 	/**
 66 	 * @name BaseClass#extend
 67 	 * @desc  为实体类添加原型方法
 68 	 * @event
 69 	 * @param {Object} obj 需要添加的原型方法
 70 	 * @example 
 71 	 * newClass.extend({
 72 	 *     key : function(arguments){}
 73 	 * }); 
 74 	 */
 75 	_class.extend = function(obj){
 76 		var extended = obj.extended;
 77 		for( var i in obj ){
 78 			_class.fn[i] = obj[i];
 79 		}
 80 		if ( extended ) {
 81 			extended( _class );
 82 		}
 83 	}
 84 	_class.proxy = function(func){
 85 		var self = this;
 86 		return(function(){
 87 			return func.apply(self, arguments);
 88 		});  
 89 	}
 90 
 91 
 92     return _class;
 93 };

 95 /**
 96  * @author 徐晨 
 97  * @name AddClass
 98  * @class 创建一个构造函数
 99  * @constructor
100  * @extends cgClass
101  * @since version 0.1 
102  * @param {String} className 函数名称
103  * @param {Object} handler 原型方法的集合
104  * @example
105  * var newClass = new cgClass.AddClass(
106  *     "ClassName",
107  *     {
108  *         init : function(options){},	
109  *         test : function(){}	
110  *     }
111  * ); 
112  */
113 cgClass.AddClass = function(className, handler){
114 	var className = className,
115 		attrs = attrs,
116 		handler = handler;
117 	function _F(arg){
118 		var _class = new cgClass.BaseClass,
119 			arg = arg;
120 		if ( !!arg && typeof arg == "object" ) {
121 			_class.attrs(arg);
122 		}
123 
124 		_class.extend(handler);
125 
126 		var _init = _class.fn.init;
127 
128 		_class.extend({
129 			// 重构默认init方法
130 			init : function(){
131 				_init.call(this, arg);
132 			},
133 			/**
134 			 * @name AddClass#get
135 			 * @desc  从内部原型中获取某属性值
136 			 * @event
137 			 * @param {String} attr 属性名称
138 			 * @example var attr = newClass.get("attrName"); 
139 			 */
140 			get : function(attr){
141 				return _class[attr];
142 			},
143 			/**
144 			 * @name AddClass#set
145 			 * @desc  设置内部原型中某属性值
146 			 * @event
147 			 * @param {String} attr 属性名称
148 			 * @param {Value} value 属性值
149 			 * @example var attr = newClass.set("attrName", "hello world!!"); 
150 			 */
151 			set : function(attr, value){
152 				_class[attr] = value;
153 			}
154 		});
155 		return new _class;
156 	}
157 	cgClass[className] = _F;
158 };
159 /**
160  * @author 徐晨 
161  * @name Create
162  * @class 创建一个新的实例
163  * @constructor
164  * @extends cgClass
165  * @since version 0.1 
166  * @param {String} className 实例名称
167  * @param {Object} arg 实例默认参数
168  * @param {Function} callback [可选] 回调函数
169  * @param {Object} callback.self 生成的实例
170  * @example var newInstance = new cgClass.Create("className", { arg : "value" }); 
171  * @example cgClass.Create(
172 	"className",
173 	{
174 		arg : "value"
175 	},
176 	function(self){
177 		self.doMethods();
178 	}
179 ); 
180  */
181 cgClass.Create = function(className, arg, callback){
182 	var className = className,
183 		_class,
184 		arg = arg || {};
185 	if ( !!cgClass[className] ){
186 		_class = new cgClass[className](arg);
187 		if ( typeof callback == "function" ) {
188 			callback( _class );
189 		}else{
190 			return _class;
191 		}
192 	}else{
193 		var _script = document.createElement("script");  
194 		_script.type= 'text/javascript';  
195 		_script.onload = _script.onreadystatechange = function() {  
196 		    if (!this.readyState || this.readyState === "loaded" || this.readyState === "complete" ) { 
197 		        _class = new cgClass[className](arg);
198 				if ( typeof callback == "function" ) {
199 					callback( _class );
200 				}
201 		        _script.onload = _script.onreadystatechange = null;  
202 		    }
203 		};  
204 		_script.src= "js/cgClass/cgClass." + className + ".js";  
205 		document.body.appendChild(_script);  
206 	}
207 };
