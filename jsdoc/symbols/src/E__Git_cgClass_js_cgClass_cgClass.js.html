 20 /**
 21  * @author 徐晨 
 22  * @name BaseClass
 23  * @class 创建基础js类
 24  * @constructor
 25  * @extends cgClass
 26  * @since version 0.1 
 27  * @param {Object} parent 可选参数，被继承类
 28  * @example var newClass = new cgClass.BaseClass(parentClass); 
 29  */
 30 cgClass.BaseClass = function(parent){
 31     var _class = function(){
 32         this.init.apply(this, arguments);
 33     };
 34     // Inherit
 35     if ( parent ) {
 36     	var subClass = function(){};
 37     	subClass.prototype = parent.prototype;
 38     	_class.prototype = new subClass;
 39     }
 40     _class.prototype.init = function(){};
 41     _class.fn = _class.prototype;
 42     _class.fn.parent = _class;
 43 	_class._super = _class.__proto__;
 44 
 45 
 46 	/**
 47 	 * @name BaseClass#attrs
 48 	 * @desc  为实体类添加属性
 49 	 * @event
 50 	 * @param {Object} obj 需要添加的属性
 51 	 * @example 
 52 	 * newClass.attrs({
 53 	 *     key : "value"
 54 	 * }); 
 55 	 */
 56 	_class.attrs = function(obj){
 57 		var included = obj.included;
 58 		for( var i in obj ){
 59 			_class[i] = obj[i];
 60 		}
 61 		if ( included ) {
 62 			included( _class );
 63 		}
 64 	}
 65 	/**
 66 	 * @name BaseClass#extend
 67 	 * @desc  为实体类添加原型方法
 68 	 * @event
 69 	 * @param {Object} obj 需要添加的原型方法
 70 	 * @example 
 71 	 * newClass.extend({
 72 	 *     key : function(arguments){}
 73 	 * }); 
 74 	 */
 75 	_class.extend = function(obj){
 76 		var extended = obj.extended;
 77 		for( var i in obj ){
 78 			_class.fn[i] = obj[i];
 79 		}
 80 		if ( extended ) {
 81 			extended( _class );
 82 		}
 83 	}
 84 
 85     return _class;
 86 };

 88 /**
 89  * @author 徐晨 
 90  * @name AddClass
 91  * @class 创建一个构造函数
 92  * @constructor
 93  * @extends cgClass
 94  * @since version 0.1 
 95  * @param {String} className 函数名称
 96  * @param {String} parent [可选] 父类函数名称
 97  * @param {Object} handler 原型方法的集合
 98  * @example
 99  * cgClass.AddClass(
100  *     "ClassName",
101  *     {
102  *         init : function(options){},	
103  *         test : function(){}	
104  *     }
105  * ); 
106  */
107 cgClass.AddClass = function(className){
108 	var className = className;
109 	if ( !!cgClass[className] ) {
110 		alert("this class is had");
111 		return false;
112 	}
113 	var	_arguments = [].slice.apply(arguments),
114 		parent,
115 		handler = _arguments.slice(-1)[0];
116 	if ( !!_arguments[1] && typeof _arguments[1] == "string" ) {
117 		parent = _arguments[1];
118 	}
119 	function _Class(arg){
120 		this.outParam = {};
121 		this.init(arg);
122 		return this;
123 	}
124 	// Inherit
125     if ( parent ) {
126     	var subClass = function(){};
127     	subClass.prototype = parent.prototype;
128     	_Class.prototype = new subClass;
129     }
130 
131     _Class.prototype.init = function(){};
132 
133     if ( !!handler && typeof handler == "object" ) {
134 	    for( var key in handler ){
135 	    	_Class.prototype[key] = handler[key];
136 	    }	
137     }
138 
139     /**
140 	 * @name AddClass#applyMethods
141 	 * @desc  生成外部调用的对象 赋值于this.outParam并返回赋值
142 	 * @event
143 	 * @param {Object} _this 当前作用于this
144 	 * @param {Object} moethods 被复制对象
145 	 * @example applyMethods({
146 	 *     attrKey : value,
147 	 *     methodKey : function(arguments){}
148 	 * }); 
149 	 */
150     _Class.prototype.applyMethods = function(_this, methods){
151     	var self = this,
152     		_methods = methods,
153     		newMethods = {};
154     	$.each(_methods, function(key, item) {
155     		_thisMethod = item;
156 	    	if ( typeof item == "function" ) {
157 	    		newMethods[key] = function(){
158 	    			item.apply(_this, arguments);
159 	    		};
160 	    	}else{
161 	    		newMethods[key] = item;
162 	    	}
163     	});
164 	    return newMethods;
165     }
166 
167 	cgClass[className] = _Class;
168 };
169 /**
170  * @author 徐晨 
171  * @name Create
172  * @class 创建一个新的实例
173  * @constructor
174  * @extends cgClass
175  * @since version 0.1 
176  * @param {String} className 实例名称
177  * @param {Object} arg 实例默认参数
178  * @param {Function} callback [可选] 回调函数
179  * @param {Object} callback.self 生成的实例
180  * @example var newInstance = new cgClass.Create(
181 	"className",
182 	{
183 		arg : "value",
184 		method : function(){}
185 	}
186 ); 
187  * @example cgClass.Create(
188 	"className",
189 	{
190 		arg : "value",
191 		method : function(){}
192 	}
193 ); 
194  * @example cgClass.Create(
195 	"className",
196 	{
197 		arg : "value"
198 	},
199 	function(self){
200 		self.doMethods();
201 	}
202 ); 
203  */
204 cgClass.Create = function(className, arg, callback){
205 	var className = className,
206 		_class,
207 		arg = arg || {};
208 	if ( !!cgClass[className] ){
209 		_class = new cgClass[className](arg);
210 		if ( typeof callback == "function" ) {
211 			callback( _class );
212 		}else{
213 			return _class;
214 		}
215 	}else{
216 		var _script = document.createElement("script");  
217 		_script.type = 'text/javascript';  
218 		_script.onload = _script.onreadystatechange = function() {  
219 		    if (!this.readyState || this.readyState === "loaded" || this.readyState === "complete" ) { 
220 		        _class = new cgClass[className](arg);
221 				if ( typeof callback == "function" ) {
222 					callback( _class );
223 				}
224 		        _script.onload = _script.onreadystatechange = null;  
225 		    }
226 		};  
227 		_script.src = "js/cgClass/cgClass." + className + ".js";  
228 		document.body.appendChild(_script);  
229 	}
230 };

232 /**
233  * @author 徐晨 
234  * @name Ajax
235  * @class Ajax方法
236  * @constructor
237  * @extends cgClass
238  * @extends jQuery
239  * @since version 0.1 
240  * @param {Object} config 配置对象
241  * @param {String} config.url 发送请求的地址
242  * @param {String} config.type ["get" : "post"] 请求方式
243 
244  * @param {String} config.dataType 请求方式
245  * @param {String} config.dataType.xml  返回 XML 文档，可用 jQuery 处理。
246  * @param {String} config.dataType.text  返回 String 字符串，可用 jQuery 处理。
247  * @param {String} config.dataType.textJson type自动转为text，返回数据预处理为 JSON 数据。
248  * @param {String} config.dataType.html 返回纯文本 HTML 信息；包含 script 元素。
249  * @param {String} config.dataType.script 返回纯文本 JavaScript 代码。不会自动缓存结果。
250  * @param {String} config.dataType.json 返回 JSON 数据 。
251  * @param {String} config.dataType.jsonp 使用 JSONP 形式调用函数时，如 "myurl?callback=?" jQuery 将自动替换 ? 为正确的函数名，以执行回调函数。
252 
253  * @param {String} config.queue ajax队列的名称
254  * @param {Number} config.timeout 设置请求超时时间（毫秒）。此设置将覆盖全局设置。
255 
256  * @param {Function} config.beforeSend 在发送请求之前调用，并且传入一个 XMLHttpRequest 作为参数
257  * @param {Function} config.success 当请求之后调用。传入返回后的数据，以及包含成功代码的字符串。
258  * @param {Function} config.error 在请求出错时调用。传入 XMLHttpRequest 对象，描述错误类型的字符串以及一个异常对象（如果有的话）
259  * @param {Function} config.complete 请求完成后回调函数 (请求成功或失败时均调用)
260 
261  * @param {Object} defData 默认参数
262 
263  * @example
264  * cgClass.Ajax({
265  *     url : "api/ajax.txt",	
266  *     data : {},	
267  *     dataType : "textJson",	
268  *     queue : "ajaxQueue",	
269  *     success : function(){},	
270  *     error : function(){}	
271  * }); 
272  * cgClass.AjaxQueueCallback.ajaxQueue = function(){}
273  */
274 cgClass.Ajax = function(config, defData){
275 	var self = this,
276 		defData = defData || {},
277 		option = $.extend(
278 			{
279 				url : "url",
280 				type: "get",
281 				dataType: "json",
282 				queue : "",
283 				timeout: (30 * 1000),
284 				beforeSend : function(xhr, ajax){},
285 				success: function( request, statusText ){},
286 				error: function( request, statusText, error ){},
287 				complete: function( request, statusText ){}
288 			},
289 			config
290 		);
291 
292 	option.data = $.extend(defData, option.data);
293 
294 	var targetSuccess = config.success,
295 		targetError = config.error,
296 		targetComplete = config.complete;
297 
298 	// 添加dataType==>textJson
299 	if ( option.dataType == "textJson" && config.success ) {
300 		option.dataType = "text";
301 		option.success = function(request, statusText){
302 			request = $.parseJSON( request );
303 			targetSuccess(request, statusText);
304 		};
305 	}
306 
307 	// ajax queue
308 	if ( !!option.queue ) {
309 		var _ajaxQueue = self.ajaxQueue[option.queue];
310 		if ( !_ajaxQueue ) {
311 			_ajaxQueue = 0;
312 		}
313 		_ajaxQueue++;
314 		self.ajaxQueue[option.queue] = _ajaxQueue;
315 		option.complete = function(request, statusText){
316 			self.ajaxQueue[option.queue]--;
317 			if ( config.complete  ) {
318 				targetComplete(request, statusText);
319 			}
320 			if (
321 				!self.ajaxQueue[option.queue]
322 				&& !!self.ajaxQueueCallback[option.queue]
323 				&& typeof self.ajaxQueueCallback[option.queue] == "function" 
324 			   )
325 			{
326 				self.ajaxQueueCallback[option.queue]();
327 			}
328 		};
329 	}
330 
331 	return $.ajax( option );			
332 };
